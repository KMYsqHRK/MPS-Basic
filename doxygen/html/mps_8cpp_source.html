282void collision() {
283 for (auto& pi : particles) {
284 if (pi.type != ParticleType::Fluid)
285 continue;
286
287 for (auto& neighbor : pi.neighbors) {
288 Particle& pj = particles[neighbor.id];
289 if (pj.type == ParticleType::Fluid && pj.id >= pi.id)
290 continue;
291
292 if (neighbor.distance < COLLISION_DISTANCE) {
293
294 double invMi = pi.inverseDensity(fluidDensity);
295 double invMj = pj.inverseDensity(fluidDensity);
296 double mass = 1.0 / (invMi + invMj);
297
298 Eigen::Vector3d normal = (pj.position - pi.position).normalized();
299 double relVel = (pj.velocity - pi.velocity).dot(normal);
300 double impulse = 0.0;
301 if (relVel < 0.0)
302 impulse = -(1 + COEFFICIENT_OF_RESTITUTION) * relVel * mass;
303 pi.velocity -= impulse * invMi * normal;
304 pj.velocity += impulse * invMj * normal;
305
306 double depth = COLLISION_DISTANCE - neighbor.distance;
307 double positionImpulse = depth * mass;
308 pi.position -= positionImpulse * invMi * normal;
309 pj.position += positionImpulse * invMj * normal;

392void setMatrix() {
393 std::vector<Eigen::Triplet<double>> triplets;
394 auto n0 = n0_forLaplacian;
395 auto a = 2.0 * DIM / (n0 * lambda);
396 coefficientMatrix.resize(particles.size(), particles.size());
397
398 for (auto& pi : particles) {
399 if (pi.boundaryCondition != FluidState::Inner)
400 continue;
401
402 double coefficient_ii = 0.0;
403 for (auto& neighbor : pi.neighbors) {
404 Particle& pj = particles[neighbor.id];
405 if (pj.boundaryCondition == FluidState::Ignored)
406 continue;
407
408 if (neighbor.distance < re_forLaplacian) {
409 double coefficient_ij = a * weight(neighbor.distance, re_forLaplacian) / fluidDensity;
410 triplets.emplace_back(pi.id, pj.id, -1.0 * coefficient_ij);
411 coefficient_ii += coefficient_ij;
412 }
413 }
414 coefficient_ii += (COMPRESSIBILITY) / (DT * DT);
415 triplets.emplace_back(pi.id, pi.id, coefficient_ii);
416 }
417 coefficientMatrix.setFromTriplets(triplets.begin(), triplets.end());

428void exceptionalProcessingForBoundaryCondition() {
429 std::vector<bool> checked(particles.size(), false);
430 std::vector<bool> connected(particles.size(), false);
431
432 for (auto& pi : particles) {
433 if (pi.boundaryCondition == FluidState::FreeSurface)
434 connected[pi.id] = true;
435 if (connected[pi.id])
436 continue;
437 // BFS for connected components
438 std::queue<size_t> queue;
439 queue.push(pi.id);
440 checked[pi.id] = true;
441 while (!queue.empty()) {
442 // pop front element
443 auto v = queue.front();
444 queue.pop();
445 // search for adjacent nodes
446 for (Eigen::SparseMatrix<double, Eigen::RowMajor>::InnerIterator it(coefficientMatrix, v); it; ++it) {
447 auto nv = it.col();
448 if (!checked[nv]) {
449 if (connected[nv]) { // connected to boundary
450 connected[v] = true;
451 checked[v] = true;
452 break;
453
454 } else {
455 queue.push(nv);
456 checked[nv] = true;
457 }
458 }
459 }
460 }

537void calPressureGradient() {
538 double a = DIM / n0_forGradient;
539
540#pragma omp parallel for
541 for (auto& pi : particles) {
542 if (pi.type != ParticleType::Fluid)
543 continue;
544
545 Eigen::Vector3d grad = Eigen::Vector3d::Zero();
546 for (auto& neighbor : pi.neighbors) {
547 Particle& pj = particles[neighbor.id];
548 if (pj.type == ParticleType::Ghost || pj.type == ParticleType::DummyWall)
549 continue;
550
551 if (neighbor.distance < re_forGradient) {
552 double w = weight(neighbor.distance, re_forGradient);
553 // double dist2 = pow(neighbor.distance, 2);
554 double dist2 = (pj.position - pi.position).squaredNorm();
555 double pij = (pj.pressure - pi.minimumPressure) / dist2;
556 grad += (pj.position - pi.position) * pij * w;
557 }
558 }
559 grad *= a;
560 pi.acceleration -= grad * pi.inverseDensity(fluidDensity);
561 }

722 if (pi.type == ParticleType::Ghost)

724

729 int iz = (int) ((pi.position.z() - domain.zMin) / bucket.length) + 1;
730
731 for (int jx = ix - 1; jx <= ix + 1; jx++) {
732 for (int jy = iy - 1; jy <= iy + 1; jy++) {
733 for (int jz = iz - 1; jz <= iz + 1; jz++) {
734 int jBucket = jx + jy * bucket.numX + jz * bucket.numX * bucket.numY;
735 int j = bucket.first[jBucket];
736
737 while (j != -1) {
738 Particle& pj = particles[j];
739
740 double dist = (pj.position - pi.position).norm();
741 if (j != pi.id && dist < reMax) {
742 pi.neighbors.emplace_back(j, dist);
743 }
744
745 j = bucket.next[j];
746 }
747 }
748 }
